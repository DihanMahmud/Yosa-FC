// reset is risky (you can loose your data parmanently so use git checkout)
git reset --soft HEAD^ (undo committing)
git reset HEAD^ (undo committing and also unstage)
git reset --hard HEAD^ (undo committing, unstage and back to previous commit)
git reset -soft HEAD-N (multiple undo committing)

git log --oneline (short info)
git show commit_id / git show HEAD~number (to show the changes on that commit)

git checkout commit_id/ HEAD~number
